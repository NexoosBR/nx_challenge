def call     

    raise StandardError, "Invalid Parameters" unless params_valid

    pmt = LoanMath.pmt(@amount.to_f, @tax.to_f, @time.to_f)
    loan = Loan.new(amount:@amount, tax:@tax, time:@time_in_months, pmt:pmt)
    loan.save!

    { loan: { id: loan.id } }
end
